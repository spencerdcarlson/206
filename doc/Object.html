def printAll
  if $EmptyFile
    puts "Sum: 0.0"
    puts "Mean: 0.0"
    puts "Median 0.0"
    exit
  else
    puts "Sum: #{sprintf("%.2f", $nums.sum)}"
    puts "Mean: #{sprintf("%.2f",$nums.mean)}"
    puts "Median: #{sprintf("%.2f",$nums.median($nums) )}" 
    exit 
  end 
end
